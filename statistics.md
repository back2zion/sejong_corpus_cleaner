    from sejong_corpus_cleaner import count_tags
    from sejong_corpus_cleaner import Corpus

    def print_counter(counter):
        sum_ = sum(counter.values())
        print('| Tag | Count | Proportion [%] |\n| --- | --- | --- |')
        for k,v in sorted(counter.items(), key=lambda x:-x[1]):
            print('| {} | {} | {:.3f} % |'.format(k, v, 100*v / sum_))

    tag_counter = count_tags(Corpus('YOURPATH'))
    print_counter(tag_counter)
